-	script	Comando Lojas	-1,{
OnInit:
	unbindatcmd "lojas";

    	set .intervalo,0;	// Tempo de intervalo para usar o comando novamente
	end;
	
OnLojas:
mes "[ Catálogo de Vendas ]";
mes "Você realmente deseja abrir o catálogo de vendas?";
mes "Aqui você pode procurar o item que está procurando para comprar dos mercadores e até comprar diretamente daqui.";
mes "Há um custo de 20,000 zeny.";
menu "Sim",sim,"Não",nao;

Sim:
next;
set .cobrarzeny,20000;
if (Zeny < .cobrarzeny) goto FALTA_ZENYS;

	searchstores 10,1;
	zeny -= .cobrarzeny;
end;

nao:
next;
mes "[ Catálogo de Vendas ]";
mes "Volte quando quiser abrir o catálogo de vendas.";
end;

FALTA_ZENYS:
dispbottom "Você não tem zeny suficiente para abrir o catálogo.";
end;

}

-	script	hold	-1,{

	OnInit:
		bindatcmd("hold", strnpcinfo(3)+"::OnHold");
		end;

	OnHold:
		if ( !@Hold ) {
			set @Hold, 1;
			message strcharinfo(0), "Modo Hold Ativo (Você esta Preso).";
			pcblockmove getcharid(3), 1;
		} else {
			set @Hold, 0;
			message strcharinfo(0), "Modo Hold Desativado (Você esta Livre).";
			pcblockmove getcharid(3), 0;
		}

}

-	script	HATEFFECT	-1,{

OnHatEffect:
	if(getgmlevel()>= 80){
	input .@number;
	if (.@number < 1) end;
	if (.@number >= 301) end;
	for(.@i = 1; .@i < 300; ++.@i)
		hateffect .@i,false;
	hateffect .@number,true;
	end;
	}

	
OnInit:
	bindatcmd "efc",strnpcinfo(3)+"::OnHatEffect";
	end;
}
